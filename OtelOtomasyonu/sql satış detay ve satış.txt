@MusteriID int,
@PersonelID int,
@OdaID int,
@OdaFiyati money,
@SatisTarihi datetime
as
begin

insert into Satis values(@MusteriID,@PersonelID,@OdaID,@OdaFiyati,@SatisTarihi)
select SCOPE_IDENTITY()--eklenen kaydýn identity deðerini dönderir. kod tarafýndada execute scalar ile bu veriyi alabiliriz.

end


go
